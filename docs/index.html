select 

const {select} = require('pg-role');
select(options: object)
Parameters
options (object)
Name Description
options.pool string (default 'default') pool connection name
options.schema string (default 'public') database schema
options.model string table to select from
options.where object conditions
options.id number add id to where conditions
options.group (array | string) (default '') group by column(s)
options.offset number (default 0) offset result index
options.limit number (default 1000) limit result length
options.deleted boolean (default false) do not filter rows with 'deleted_at' timestamp
Example
async function getEmployee (id) {
     return await select({
         model: 'employees',
         id
     });
 }
async function getDeletedEmployee (id) {
     return await select({
         model: 'employees',
         id,
         where: {
             deleted_at: 'not null'
         }
     });
 }
async function getActiveEmployeesByRole (role) {
     return await select({
         model: 'employees',
         where: {
             role
         }
     });
 }
async function getAllEmployeesByRole (role) {
     return await select({
         model: 'employees',
         deleted: true,
         where: {
             role
         }
     });
 }

     `);

     return await query({
